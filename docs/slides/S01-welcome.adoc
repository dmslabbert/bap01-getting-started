== Some SQL Code
[source,sql,linenums]
----
CREATE TABLE employees (
    id          INT PRIMARY KEY AUTO_INCREMENT,
    first_name  VARCHAR(50) NOT NULL,
    last_name   VARCHAR(50) NOT NULL,
    email       VARCHAR(100) UNIQUE,
    hired_on    DATE,
    salary      DECIMAL(10, 2)
);

INSERT INTO employees (first_name, last_name, email, hired_on, salary)
VALUES
    ('Alice', 'Smith',  'alice@example.com',  '2021-03-15', 75000.00),
    ('Bob',   'Jones',  'bob@example.com',    '2022-07-01', 68000.00),
    ('Carol', 'White',  'carol@example.com',  '2023-01-10', 82000.00);

SELECT
    e.first_name,
    e.last_name,
    e.salary
FROM employees e
WHERE e.salary > 70000
ORDER BY e.salary DESC;
----
